Greatest Common Divisor Algorithm

Greatest Common Divisor Algorithm

#include <cstdio>


int gcd(int a, int b) {
    int c = a % b;
    if (c != 0) {
        return gcd(b, c);
    } else {
        return b;
    }

}

int main() {
    int a, b;
    scanf("%d %d", &a, &b);

    if (a >= b) {
        printf("%d\n", gcd(a, b));
    } else {
        printf("%d\n", gcd(b, a));
    }

    return 0;
}
